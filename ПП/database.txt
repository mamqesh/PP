CREATE TABLE [Unit] (
	UnitID integer PRIMARY KEY NOT NULL,
	UnitName nvarchar(50) NOT NULL
)
GO
CREATE TABLE [Image] (
	ImageID integer PRIMARY KEY NOT NULL,
	[Image] image NOT NULL,

)
GO
CREATE TABLE [Product] (
	ProductID integer NOT NULL,
	ProductNumber nvarchar(25) NOT NULL,
	PRIMARY KEY (ProductID,ProductNumber),
	ProductName nvarchar(100) NOT NULL,
	[Count] nvarchar(100) NOT NULL,
	Unit integer NOT NULL,
	ProductNote nvarchar(255),
	[Image] integer,
	FOREIGN KEY (Unit) REFERENCES Unit(UnitID),
	FOREIGN KEY ([Image]) REFERENCES [Image](ImageID)
)
GO
